docker run --name curd_api_rust_write -e SERVER_HOST=0.0.0.0 -e SERVER_PORT=8000 -e HTTPS=false -e DATABASE_URL=${DATABASE_URL} -e DATABASE_MAX_CONNECTIONS=30 -e DATABASE_MIN_CONNECTIONS=2 -e JWT_SECRET=${JWT_PWD} -e JWT_EXPIRES_IN=3600 -e JWT_REFRESH_IN=86400 -e ENCRYPTION_ALGORITHM=aes-256-gcm -e ENCRYPTION_KEY_LENGTH=32 -e ENCRYPTION_ITERATIONS=100000 -e SERVICE_ROLE=write -e SERVICE_ID=crud-write-01 -e ALLOWED_TABLES=users,resources,encryption_keys -e RUN_MIGRATIONS=true -e MIGRATION_STRATEGY=${MIGRATION_STRATEGY} -p 7981:8000 --restart unless-stopped ghcr.io/aspnmy/curd_apirust:latest
